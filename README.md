# n8n en Railway
Usa PostgreSQL de Railway y variables de entorno:
- N8N_HOST
- WEBHOOK_URL
- N8N_EDITOR_BASE_URL
- N8N_ENCRYPTION_KEY
- N8N_BASIC_AUTH_ACTIVE=true
- N8N_BASIC_AUTH_USER
- N8N_BASIC_AUTH_PASSWORD
- DB_TYPE=postgresdb
- DB_POSTGRESDB_HOST
- DB_POSTGRESDB_PORT
- DB_POSTGRESDB_USER
- DB_POSTGRESDB_PASSWORD
- DB_POSTGRESDB_DATABASE
- N8N_TELEMETRY_DISABLED=true
- N8N_DIAGNOSTICS_DISABLED=true
- N8N_PORT=5678
